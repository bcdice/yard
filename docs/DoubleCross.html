def check_nD10(total_n, _dice_n, signOfInequality, diff, dice_cnt, _dice_max, n1, _n_max) # ゲーム別成功度判定(nD10)
  return '' unless signOfInequality == ">="

  if n1 >= dice_cnt
    return " ＞ ファンブル"
  elsif total_n >= diff
    return " ＞ 成功"
  else
    return " ＞ 失敗"
  end
end

def rollDiceCommand(command)
  if (dx = parse(command))
    return execute_dx(dx)
  end

  if command == 'ET'
    return get_emotion_table
  end

  return nil
end
