def formatted_modifier
  if modifier == 0
    ''
  elsif modifier > 0
    "+#{modifier}"
  else
    modifier.to_s
  end
end

def to_s
  lhs = "#{num}DX#{critical_value}#{formatted_modifier}"

  return target_value ? "#{lhs}>=#{target_value}" : lhs
