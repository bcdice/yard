def rollAct(counts, judgment = 0, critical = 0, fumble = 0)
  if critical < 1
    critical = 1
  end
  if judgment <= 0
    judgment = critical
  elsif critical > judgment
    critical = judgment
  end

  if fumble <= 0
    fumble = 20
  end
  if counts <= 0
    counts = 1
    fumble -= 5
  end
  if fumble < 2
    fumble = 2
  elsif fumble > 20
    fumble = 20
  end

  value, string = roll(counts, 20, 1)
  text = "(#{counts}A#{judgment}C#{critical}F#{fumble}) ＞ #{string} ＞ "

  unless counts == 1
    value = string.split(",").map(&:to_i).min
    text += "#{value} ＞ "
  end

  if value >= fumble
    text += 'ファンブル'
  elsif value <= critical
    text += 'クリティカル'
  elsif value > judgment
    text += '失敗'
  else
    text += '成功'
  end
  return text
end

def tableText(title, table, sign = '')
  number, = roll(1, 20)
  index = number
  if sign == '+'
    index += 1
  elsif sign == '-'
    index -= 1
  end

  text = "#{title} ＞ #{index}"
  unless sign.empty?
    text += "[#{number}#{sign}1]"
  end
  return text + ' ＞ ' + table[index]
end
