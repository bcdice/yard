def roll(bcdice)
  dice1, = bcdice.roll(1, 6)
  dice2, = bcdice.roll(1, 6)

  index1 = dice1 - 1
  index2 = dice2 - 1
  return "#{@name}(#{dice1}#{dice2}) ＞ #{@items[index1][index2]}"
end
