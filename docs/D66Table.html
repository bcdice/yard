def roll(bcdice)
  dice = Array.new(2) do
    val, = bcdice.roll(1, 6)
    val
  end

  dice.sort!
  if @sort_type == :desc
    dice.reverse!
  end

  key = dice[0] * 10 + dice[1]
  return "#{@name}(#{key}) ＞ #{@items[key]}"
end
