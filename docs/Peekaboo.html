def check_2D6(total_n, dice_n, signOfInequality, diff, _dice_cnt, _dice_max, _n1, _n_max)
  return '' unless signOfInequality == ">="

  if dice_n <= 2
    return " ＞ ファンブル(【眠気】が1d6点上昇)"
  elsif dice_n >= 12
    return " ＞ スペシャル(【魔力】あるいは【眠気】が1d6点回復)"
  elsif total_n >= diff
    return " ＞ 成功"
  else
    return " ＞ 失敗"
  end
end
