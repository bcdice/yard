def eval(game_system, randomizer)
  times = @times.eval(game_system, nil)
  sides = eval_sides(game_system)

  dice_list = randomizer.roll(times, sides)

  total = dice_list.sum()
  @text = "#{total}[#{dice_list.join(',')}]"

  return total
end

def expr(game_system)

  "#{times}D#{sides}"
end
