def valid?(sides)
  return false unless @threshold

  case @cmp_op
  when :<=
    @threshold < sides
  when :<
    @threshold <= sides
  when :>=
    @threshold > 1
  when :>
    @threshold >= 1
  when :'!='
    (1..sides).include?(@threshold)
  else # :==
    true
  end
end
