def check_1D100(total, _dice_total, cmp_op, target)
  return '' if target == '?'
  return '' unless cmp_op == :<=

  t10 = total / 10
  d10 = target / 10
  sl = d10 - t10

  if total <= 5 && sl < 1
    # 自動成功時のSLは最低でも1
    sl = 1
  elsif total >= 96 && sl > -1
    # 自動失敗時のSLは最大でも-1
    sl = -1
  end

  result =
    if total <= 5
      "自動成功"
    elsif total >= 96
      "自動失敗"
    elsif total <= target
      "成功"
    else
      "失敗"
    end

  sl_text = format("(SL%+d)", sl)

  " ＞ #{result}#{sl_text} ＞ #{result_detail(sl, total, target)}"
end
