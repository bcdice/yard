def brand_only()
  DiceTable::D66Table.new(
    @name,

    @items.transform_values { |e| e.split("\n").first }
  )
end

def roll(randomizer)
  value = randomizer.roll_d66(D66SortType::ASC)
  "#{@name}(#{value}) ＞ #{@items[value]}"
end
