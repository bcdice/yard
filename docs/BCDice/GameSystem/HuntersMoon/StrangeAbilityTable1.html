def roll(randomizer)
  dice1 = randomizer.roll_once(6)
  dice2 = randomizer.roll_once(6)
  index = (dice1 - 1) * 6 + (dice2 - 1)

  @dice = dice1 * 10 + dice2
  @chosen = BODY[index]
end
