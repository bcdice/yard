def result_nd6(total, _dice_total, dice_list, cmp_op, target)
  return nil unless cmp_op == :>=

  result = Result.new
  sequence = []

  if dice_list.count(6) >= 2
    total += 10
    result.critical = true
    sequence.push("（クリティカル）", total.to_s)
  end

  if target != '?'
    if total >= target
      sequence.push("成功")
      result.success = true
    else
      sequence.push("失敗")
      result.failure = true
    end
  end

  if sequence.empty?
    return nil
  end

  result.text = sequence.join(" ＞ ")
  return result
end
