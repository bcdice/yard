def eval_game_system_specific_command(command)
  if (m = /^(\d+)?DT$/i.match(command))
    count = m[1]&.to_i || 2
    return roll_dt(command, count)
  elsif (m = /^(\d+)?AS$/i.match(command))
    count = m[1]&.to_i || 2
    return roll_as(command, count)
  end

  return roll_tables(command, self.class::TABLES)
end
