def roll(randomizer)
  dice = randomizer.roll_once(10)
  chosen = @items[dice - 1]

  result = "#{@name}(#{dice}) ＞ #{chosen[:element]} （意味：#{chosen[:mean]}）"
  return result, chosen[:element]
end
