def eval_game_system_specific_command(command)
  info = TABLES[command.upcase]
  return nil if info.nil?

  name = info[:name]
  type = info[:type]
  table = info[:table]

  text, number =
    case type
    when '1D20'
      get_table_by_nDx(table, 1, 20)
    end

  return nil if text.nil?

  return "#{name}(#{number}) ＞ #{text}"
end
