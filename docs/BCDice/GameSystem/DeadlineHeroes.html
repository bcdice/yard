def eval_game_system_specific_command(command)
  case command
  when /^DLH/i
    resolute_action(command)
  when /^DC\w/i
    roll_death_chart(command)
  when 'HNC'
    roll_hero_name_chart()
  else
    roll_tables(command, TABLES)
  end
end
