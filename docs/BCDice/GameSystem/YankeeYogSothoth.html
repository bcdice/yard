def result_2d6(_total, dice_total, _dice_list, cmp_op, _target)
  return nil unless cmp_op == :>=

  if dice_total <= 2
    Result.fumble("ファンブル(判定失敗。ファンブル表（FT）を振ること)")
  elsif dice_total >= 12
    Result.critical("スペシャル(判定成功。【テンション】が１段階上昇)")
  end
end
