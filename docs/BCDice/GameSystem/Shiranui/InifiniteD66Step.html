def score
  if repdigit?
    # ゾロ目の場合

    digit = @dices.first

    if digit == 1
      # 1 のゾロ目なら 0 となる
      0
    else
      # 1 以外のゾロ目なら、数字の 10 倍となる
      digit * 10
    end
  else
    # ゾロ目でない場合は、 D66 様式で値を算出する
    @dices[0] * 10 + @dices[1]
  end
end
