def roll(randomizer)
  dice = randomizer.roll_once(@items.length)
  chosen = @items[dice - 1]

  critical = chosen == CRITICAL

  result_tail =
    if critical
      "マジック"
    elsif !chosen.empty?
      "成功"
    else
      "失敗"
    end

  Result.new.tap do |r|
    r.critical = critical
    r.condition = !chosen.empty?
    r.text = [
      "(D#{@items.length})",
      dice.to_s,
      chosen.empty? ? nil : chosen,
      result_tail,
    ].compact.join(" ＞ ")
  end
end
