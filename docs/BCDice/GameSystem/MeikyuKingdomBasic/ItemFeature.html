def choice(randomizer)
  dice = randomizer.roll_sum(@times, @sides)
  index = dice - @times

  chosen = @items[index]
  chosen = chosen.choice(randomizer) if chosen.respond_to?(:choice)

  return "[#{dice}]#{chosen}"
end
