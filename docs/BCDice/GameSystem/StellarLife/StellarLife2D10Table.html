def roll(randomizer)
  dices = randomizer.roll_barabara(2, 10)

  row_index = dices[0] == 10 ? 0 : dices[0]
  column_index = dices[1] == 10 ? 0 : dices[1]

  key = row_index * 10 + column_index
  chosen = @items[key]

  return RollResult.new(@name, format("%02d", key), chosen)
end
