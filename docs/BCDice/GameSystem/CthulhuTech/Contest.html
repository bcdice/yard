def result_str(success, _fumble, _critical, diff)
  formatted = super

  if success
    damage_roll_num = (diff / 5.0).ceil
    damage_roll = "#{damage_roll_num}D10"

    "#{formatted}（ダメージ：#{damage_roll}）"
  else
    formatted
  end
end
