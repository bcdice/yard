def pick_item(index)
  if index <= 6
    translate("LogHorizon.TRS.below_lower_limit", value: 6)
  elsif index <= 53
    @items[index]
  else
    translate("LogHorizon.TRS.exceed_upper_limit", value: 54)
  end
end
