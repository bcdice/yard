def roll(randomizer, under_ganken_bonus)
  dice = randomizer.roll_once(100)
  if under_ganken_bonus
    dice = (dice - 20).clamp(1, 100)
  end

  chosen = @items.find { |key, _| key >= dice }[1]

  "#{@name}(#{dice}) ï¼ž #{chosen}"
end
