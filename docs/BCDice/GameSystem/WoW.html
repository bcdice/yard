def eval_game_system_specific_command(command)
  case command
  when 'GG'
    return roll_gg
  when /^GG([A-H])$/
    return roll_table(::Regexp.last_match(1))
  when 'FT'
    return roll_fumble_table
  else
    return roll_wow(command)
  end
end
