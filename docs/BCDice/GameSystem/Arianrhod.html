def check_nD6(total, _dice_total, dice_list, cmp_op, target)
  n_max = dice_list.count(6)

  if dice_list.count(1) == dice_list.size
    # 全部１の目ならファンブル
    " ＞ ファンブル"
  elsif n_max >= 2
    # ２個以上６の目があったらクリティカル
    " ＞ クリティカル(+#{n_max}D6)" if n_max >= 2
  elsif cmp_op != :>= || target == '?'
    ''
  elsif total >= target
    " ＞ 成功"
  else
    " ＞ 失敗"
  end
end
