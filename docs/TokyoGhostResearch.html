def getCheckResult(command)
  output = ""
  diff = 0

  if /TK?<=(\d+)/i =~ command
    diff = Regexp.last_match(2).to_i
  end

  if diff > 0
    output += "(1D10<=#{diff})"

    total_n, = roll(1, 10)
    output += ' ＞ ' + total_n.to_s
    output += ' ＞ ' + getCheckResultText(total_n, diff)
  end
  return output
end

def rollDiceCommand(command)
  output =
    case command.upcase

    when /TK/i
      return getCheckResult(command)

    when 'OP'
      tgr_opening_table
    when 'TB'
      tgr_common_trouble_table
    end

  return output
end
