def dynamic_load(id)
  class_name = id.tr(":.", "_")

  # 対象ディレクトリの外にあるファイルをロードされないように制約を設ける
  unless /\A[A-Z]\w*\z/.match?(class_name)
    return nil
  end

  require "bcdice/game_system/#{class_name}"

  return BCDice::GameSystem.const_get(class_name)
rescue LoadError, NameError
  return nil
end
