def parse()
  lhs, cmp_op, rhs = @expr.partition(/[<>=]+/)

  cmp_op = Normalize.comparison_operator(cmp_op)
  if !rhs.empty? && rhs != "?"
    rhs = ArithmeticEvaluator.new.eval(rhs)
  end

  @tokens = tokenize(lhs)
  lhs = expr()

  if @idx != @tokens.size || !@contain_dice_roll
    @error = true
  end

  return AddDice::Node::Command.new(lhs, cmp_op, rhs)
end
