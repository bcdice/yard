def analize
  isAnalized = false

  @args.each do |arg|
    result = analizeArg(arg)
    checkArg(arg)

    if result
      isAnalized = true
    end
  end

  return isAnalized
end

def analizeArg(arg)
  return false unless /^-([scngmeir])(.+)$/i =~ arg

  command = Regexp.last_match(1).downcase
  @param = Regexp.last_match(2)

  case command
  when "s"
    setServer
  when "c"
    setChannel
  when "n"
    setNick
  when "g"
    setGame
  when "m"
    setMessageSendType
  when "e"
    readExtraCard
  when "i"
    setIrcServerCharacterCode
  else
    return false
  end

  return true
end
